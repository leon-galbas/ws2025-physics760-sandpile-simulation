\begin{frame}{Introduction}

    \textbf{Criticality:}
    \begin{itemize}
        \item Correlations span the whole system
        \item Small perturbations can affect the whole system
        \item Scale invariance $\implies$ power laws
    \end{itemize}

\end{frame}

\begin{frame}{Self-organized criticality}
    \begin{itemize}
        \item \textbf{Ising model:} Needs a fine-tuned parameter $T_c$ to reach criticality.
        \item \textbf{Earthquakes:} Exhibit criticality on their own. $\implies$ Self-Organized Criticality (SOC)
        \item SOC gives an explanation approach for the widespread appearance of power laws in nature.
    \end{itemize}

\end{frame}

\begin{frame}{Why sandpiles?}

    \small

    \begin{itemize}
        \item The Bak-Tang-Wiesenfeld (BTW) sandpile model is a very simple model to study SOC (\cite{bak1987self})
        \item Analogy: Slowly dropping sand onto a table causes regular avalanches of various sizes
    \end{itemize}

    \begin{figure}
        \centering
        \includegraphics[width=\linewidth,height=0.55\textheight,keepaspectratio]{sandpile_example}
        \caption{Example state of a two dimensional BTW sandpile model.}
    \end{figure}

\end{frame}



\begin{frame}{Goals of our project}

    \begin{itemize}
        \item Simulate a BTW sandpile cellular automaton in $d=2,3,4$ dimensions
        \item Show that the system exhibits self-organized criticality
        \item Confirm the emergence of power laws and compute the scaling exponents
        \item Compare the results for different hyperparameters and dimensions
    \end{itemize}

\end{frame}
