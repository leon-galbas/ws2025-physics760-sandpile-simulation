\begin{frame}{How to detect stationarity?}
    \pause
    \begin{enumerate}
        \item Initialize the sandpile.
        \item Define a threshold $\eps>0$.
        \item Track $\expval{z}$ every $\num{1000}$ steps while continually perturbing and relaxing the sandpile.
        \item Do a linear fit through the last $\num{50}$ values.
        \item Stop, if the slope is smaller than $\eps$.
    \end{enumerate}
\end{frame}

\begin{frame}{Determining the scaling region}
    \begin{enumerate}
        \item Apply linear regression on rolling window of fixed size.
        \item Determine the slope $m_i$ and uncertainty $\Delta m_i$ for each window.
        \item Calculate differences\begin{equation*}
            d_i = \abs{m_i-m_{i+1}}, \qquad
            \Delta d_i = \sqrt{(\Delta m_i)^2+(\Delta m_{i+1})^2}.
        \end{equation*}
        \item Classify as stable, if $d_i < k\cdot\Delta d_i$, and if both fits satisfy $r_i^2 \geq r_{\mathrm{thresh}}$.
        \item Set upper and lower bounds of scaling region.
    \end{enumerate}
\end{frame}

\begin{frame}{Error estimation of scaling exponents}
    \pause 
    \begin{enumerate}
        \item compute upper and lower bounds of scaling region
        \item vary the boundaries by $\pm 3$
        \item compute the slope of every valid pair of boundaries
        \item compute standard deviation  $\sigma_\f{sys}$ of all slopes
        \item combine with linear fit error $\sigma_\f{stat}$
    \end{enumerate}
    \pause 

    \begin{equation*}
    \sigma_\f{tot}=\sqrt{\sigma_\f{stat}^{2}+\sigma_\f{sys}^{2}}
    \end{equation*}
\end{frame}
